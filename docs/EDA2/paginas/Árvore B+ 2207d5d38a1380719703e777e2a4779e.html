Árvore B+

A Árvore B+ é uma estrutura de dados autobalanceada, derivada da Árvore B, otimizada para armazenamento e recuperação eficiente de dados, especialmente em memória secundária (ex: bancos de dados, sistemas de arquivos).

Estrutura e Características

  • Chaves apenas em folhas

    Os dados reais ou ponteiros para os dados são armazenados somente nos nós folha.

  • Índices com chaves duplicadas

    Os nós internos (índices) armazenam somente as chaves de busca, sem dados associados.

  • Ligação sequencial entre folhas

    As folhas estão encadeadas, permitindo acesso sequencial em O(1).

  • Separação entre dados e índices

    Índices e dados podem ser armazenados em arquivos distintos.

  • Ordem da árvore (M)
    • Cada nó (exceto raiz) armazena entre M/2 e M–1 chaves, para M > 3.
    • Garante balanceamento e eficiência.
  • Complexidade
    • Busca aleatória: O(log n)
    • Inserção: O(log n)
    • Remoção: O(log n)

Inserção (Resumo do Processo)

  1. Localiza-se a folha correta.
  1. Insere-se a chave na posição adequada.
  1. Se houver transbordo, ocorre divisão da página:
    • As M–1 chaves são divididas.
    • A menor chave da nova folha direita é copiada para o nó pai.
    • Diferente da Árvore B, a chave não é removida da folha, apenas copiada.

Considerações de Implementação

  • Três arquivos recomendados:
    1. Metadados: ponteiro da raiz, flag de folha.
    1. Índices: nós internos, com chaves e ponteiros.
    1. Dados: nós folha com registros e ponteiros sequenciais.
  • Cálculo da ordem mínima (d):

    d=(Bk)−1d = \left(\frac{B}{k}\right) - 1

    Onde:

    • B = tamanho do bloco
    • k = tamanho da chave

Aplicações

  • Bancos de Dados: SQL Server, Oracle, PostgreSQL, MariaDB, SQLite.
  • Sistemas de Arquivos: ReiserFS, XFS, JFS2.
  • Gerência de Dados: CouchDB, Tokyo Cabinet, Tokyo Tyrant.

Algoritmos

Nós

typedef struct _no{
		bool folha;
		int total;
		int *chaves;
		struct _no **filhos;
		struct _no *prox;
} No;

Árvore 

struct {
		No *raiz;
		int ordem; 
}ArvoreBMais;

Cria nó

No* criaNo(int ordem, bool folha) {
    No* novo = (No*) malloc(sizeof(No));
    novo->folha = folha;
    novo->total = 0;
    novo->chaves = (int*) malloc((ordem - 1) * sizeof(int));
    novo->filhos = (No**) malloc(ordem * sizeof(No*));
    for (int i = 0; i < ordem; i++) {
        novo->filhos[i] = NULL;
    }
    novo->prox = NULL;
    return novo;
}

Cria Árvore

ArvoreBMais* criaArvore(int ordem) {
    ArvoreBMais* arv = (ArvoreBMais*) malloc(sizeof(ArvoreBMais));
    arv->ordem = ordem;
    arv->raiz = criaNo(ordem, true);
    return arv;
}

Busca

bool busca(No* no, int chave) {
    int i = 0;
    while (i < no->total && chave > no->chaves[i]) i++;

    if (no->folha) {
        return (i < no->total && no->chaves[i] == chave);
    } else {
        return busca(no->filhos[i], chave);
    }
}

Inserção

void divideFolha(ArvoreBMais* arv, No* pai, int pos, No* folha) {
    int ordem = arv->ordem;
    No* nova = criaNo(ordem, true);
    int meio = (ordem - 1) / 2;

    nova->total = folha->total - meio;
    for (int i = 0; i < nova->total; i++) {
        nova->chaves[i] = folha->chaves[meio + i];
    }

    folha->total = meio;
    nova->prox = folha->prox;
    folha->prox = nova;

    for (int i = pai->total; i > pos; i--) {
        pai->chaves[i] = pai->chaves[i - 1];
        pai->filhos[i + 1] = pai->filhos[i];
    }

    pai->chaves[pos] = nova->chaves[0];
    pai->filhos[pos + 1] = nova;
    pai->total++;
}

void divideInterno(ArvoreBMais* arv, No* pai, int pos, No* filho) {
    int ordem = arv->ordem;
    No* novo = criaNo(ordem, false);
    int meio = (ordem - 1) / 2;

    novo->total = filho->total - meio - 1;
    for (int i = 0; i < novo->total; i++) {
        novo->chaves[i] = filho->chaves[meio + 1 + i];
    }
    for (int i = 0; i <= novo->total; i++) {
        novo->filhos[i] = filho->filhos[meio + 1 + i];
    }

    for (int i = pai->total; i > pos; i--) {
        pai->chaves[i] = pai->chaves[i - 1];
        pai->filhos[i + 1] = pai->filhos[i];
    }

    pai->chaves[pos] = filho->chaves[meio];
    pai->filhos[pos + 1] = novo;
    pai->total++;

    filho->total = meio;
}

void insereRec(ArvoreBMais* arv, No* no, int chave, No** novoFilho, int* chavePromovida) {
    int i = 0;
    while (i < no->total && chave > no->chaves[i]) i++;

    if (no->folha) {
        for (int j = no->total; j > i; j--) {
            no->chaves[j] = no->chaves[j - 1];
        }
        no->chaves[i] = chave;
        no->total++;

        if (no->total == arv->ordem) {
            *novoFilho = criaNo(arv->ordem, true);
            int meio = (arv->ordem - 1) / 2;

            (*novoFilho)->total = no->total - meio;
            for (int j = 0; j < (*novoFilho)->total; j++) {
                (*novoFilho)->chaves[j] = no->chaves[meio + j];
            }
            (*novoFilho)->prox = no->prox;
            no->prox = *novoFilho;
            no->total = meio;
            *chavePromovida = (*novoFilho)->chaves[0];
        } else {
            *novoFilho = NULL;
        }

    } else {
        No* filho = no->filhos[i];
        No* novo = NULL;
        int promovida;
        insereRec(arv, filho, chave, &novo, &promovida);

        if (novo != NULL) {
            for (int j = no->total; j > i; j--) {
                no->chaves[j] = no->chaves[j - 1];
                no->filhos[j + 1] = no->filhos[j];
            }
            no->chaves[i] = promovida;
            no->filhos[i + 1] = novo;
            no->total++;

            if (no->total == arv->ordem) {
                *novoFilho = criaNo(arv->ordem, false);
                int meio = (arv->ordem - 1) / 2;
                (*novoFilho)->total = no->total - meio - 1;

                for (int j = 0; j < (*novoFilho)->total; j++) {
                    (*novoFilho)->chaves[j] = no->chaves[meio + 1 + j];
                    (*novoFilho)->filhos[j] = no->filhos[meio + 1 + j];
                }
                (*novoFilho)->filhos[(*novoFilho)->total] = no->filhos[no->total];
                *chavePromovida = no->chaves[meio];
                no->total = meio;
            } else {
                *novoFilho = NULL;
            }
        } else {
            *novoFilho = NULL;
        }
    }
}

void inserir(ArvoreBMais* arv, int chave) {
    No* novo = NULL;
    int promovida;
    insereRec(arv, arv->raiz, chave, &novo, &promovida);

    if (novo != NULL) {
        No* novaRaiz = criaNo(arv->ordem, false);
        novaRaiz->chaves[0] = promovida;
        novaRaiz->filhos[0] = arv->raiz;
        novaRaiz->filhos[1] = novo;
        novaRaiz->total = 1;
        arv->raiz = novaRaiz;
    }
}
